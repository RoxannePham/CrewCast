DATABASE SCHEMA (PRISMA)
Implement exactly the following models (copy unchanged):

model User {
id String @id @default(uuid())
name String
email String @unique
password String
rating Float?
reliabilityScore Float?
roles String[]
skills String[]
portfolio String[]
avatar String?
location String?
events Event[] @relation("HostEvents")
applications Application[]
messages Message[]
}

model Event {
id String @id @default(uuid())
hostId String
host User @relation("HostEvents", fields: [hostId], references: [id])
title String
date DateTime
time String
location String
coverImage String?
themeColor String?
description String?
roles Role[]
}

model Role {
id String @id @default(uuid())
eventId String
event Event @relation(fields: [eventId], references: [id])
roleType String
pay Int?
shiftStart String?
shiftEnd String?
slots Int
applications Application[]
}

model Application {
id String @id @default(uuid())
userId String
roleId String
status String
note String?
timestamp DateTime @default(now())
user User @relation(fields: [userId], references: [id])
role Role @relation(fields: [roleId], references: [id])
}

model Message {
id String @id @default(uuid())
chatId String
senderId String
text String?
imageUrl String?
timestamp DateTime @default(now())
sender User @relation(fields: [senderId], references: [id])
}
